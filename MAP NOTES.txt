MAP KEY:

# Blocks
W = Wall
O = Win

# Characters:
R = Red
G = Green
B = Blue

# [Block - Switch]:    
# note: These are adjacent to their Colour's letter key (which is reserved for character start positions)
[T - t] = Red
[H - h] = Green
[N - n] = Blue
# Mixed
[Y - y] = Yellow
[C - C] = Cyan
[M - m] = Magenta
